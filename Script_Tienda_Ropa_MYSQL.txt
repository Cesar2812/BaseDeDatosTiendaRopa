CREATE DATABASE tienda_ropa_ebra;

-- Usar la base de datos
USE tienda_ropa_ebra;

-- Tabla  Clientes
CREATE TABLE Clientes (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(50),
    Apellido VARCHAR(50),
    Email VARCHAR(100)
);

-- Tabla Productos
CREATE TABLE Productos (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(100),
    Descripcion TEXT,
    Precio DECIMAL(10, 2)
);

-- Tabla  Categor√≠as de Productos
CREATE TABLE Categorias (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(50)
);

-- Tabla Pedidos
CREATE TABLE Pedidos (
    ID INT PRIMARY KEY,
    ClienteID INT,
    FechaPedido DATE,
    FOREIGN KEY (ClienteID) REFERENCES Clientes(ID)
);

-- Tabla Detalles de Pedido
CREATE TABLE DetallesPedido (
    ID INT PRIMARY KEY,
    PedidoID INT,
    ProductoID INT,
    Cantidad INT,
    FOREIGN KEY (PedidoID) REFERENCES Pedidos(ID),
    FOREIGN KEY (ProductoID) REFERENCES Productos(ID)
);

-- Tabla Empleados
CREATE TABLE Empleados (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(50),
    Apellido VARCHAR(50),
    Puesto VARCHAR(50)
);

-- Tabla Proveedores
CREATE TABLE Proveedores (
    ID INT PRIMARY KEY,
    NombreEmpresa VARCHAR(100),
    Contacto VARCHAR(100),
    Telefono VARCHAR(20)
);

CREATE TABLE OrdenesCompra (
    ID INT PRIMARY KEY,
    ProveedorID INT,
    FechaOrden DATE,
    FOREIGN KEY (ProveedorID) REFERENCES Proveedores(ID)
);

-- Tabla Compras
CREATE TABLE Compras (
    ID INT PRIMARY KEY,
    ProveedorID INT,
    FechaCompra DATE,
    FOREIGN KEY (ProveedorID) REFERENCES Proveedores(ID)
);


-- Tabla Detalles de Compra
CREATE TABLE DetallesCompra (
    ID INT PRIMARY KEY,
    CompraID INT,
    ProductoID INT,
    Cantidad INT,
    PrecioUnitario DECIMAL(10, 2),
    FOREIGN KEY (CompraID) REFERENCES Compras(ID),
    FOREIGN KEY (ProductoID) REFERENCES Productos(ID)
);

-- Tabla Inventario
CREATE TABLE Inventario (
    ProductoID INT PRIMARY KEY,
    CantidadStock INT,
    FOREIGN KEY (ProductoID) REFERENCES Productos(ID)
);

-- Tabla Ventas
CREATE TABLE Ventas (
    ID INT PRIMARY KEY,
    ClienteID INT,
    FechaVenta DATE,
    FOREIGN KEY (ClienteID) REFERENCES Clientes(ID)
);

-- Tabla detalle de venta
CREATE TABLE DetallesVenta (
    ID INT PRIMARY KEY,
    VentaID INT,
    ProductoID INT,
    Cantidad INT,
    PrecioUnitario DECIMAL(10, 2),
    FOREIGN KEY (VentaID) REFERENCES Ventas(ID),
    FOREIGN KEY (ProductoID) REFERENCES Productos(ID)
);
